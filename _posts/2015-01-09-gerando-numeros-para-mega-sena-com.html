{% highlight bash %}#!/bin/bash

# ----------------------------------------------------------------------------

# Autor: Marcos da B.M. Oliveira, http://www.terminalroot.com.br/

# Gerando Números para Mega-Sena com Shell Script

# ./megasena.sh

# Desde: Sex 09 Jan 2015 18:25:32 BRST

# Versão: 1

# Licença: GPL

# ----------------------------------------------------------------------------

#criamos a função

function megasena(){

    #formamos um loop para gerar  números

    for ((i = 0; i <= 6; i++))

        do   

            # O -en é para gerar na mesma linha e gerar cores

            # configuramos a cor de fundo desejada e chamamos de 0 a 60 a variável $RANDOM   

            echo -en " \033[42;1;37m $(($RANDOM % 61)) \033[0m"           

    done

}

# titulo com cor azul

echo  -e " \033[1;34m Números gerados:  \033[0m"

# chamamos a função

megasena

#pulamos uma linha

echo{% endhighlight %}
